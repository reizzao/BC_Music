// 2 - Interno: O Cli precisa usar algum recurso e pra isso criamos uma função
function RecursoCli (d) {
  const i = new DepRecursoCliClass()
  return i.Op_Ola(d.dep1)
}

// 1 - Externo: Ofertamos ao Cli uma porta de entrada um objeto para ele usar o Recurso.
const Request = {
  dep1 : "Rei1"
}

// 3 - interno : Onde seria uma função pode ser um método de uma classe , evitaria hell de funções e abriria opções de operações.
class DepRecursoCliClass {
  Op_Ola (d) {
    return `Ola ${d}`
  }

  Op_ComoVai (d) {
    return `Ola ${d}`
  }

}

// 1 - Interno : Testando o Recurso Ofertado.
console.log( RecursoCli ( Request ) );
